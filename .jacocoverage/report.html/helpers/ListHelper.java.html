package helpers;

import java.util.HashSet;
import java.util.List;
import java.util.Set;

public class ListHelper {
    public static boolean isListOfSameElements(List<? extends Object> l) {
        Set<Object> set = new HashSet<>(l.size());
        for (Object o : l) {
            if (set.isEmpty()) {
                set.add(o);
            } else {
                if (set.add(o)) {
                    return false;
                }
            }
        }
        return true;
    }
}
