package nash;

import java.util.Objects;

/**
 *
 * @author Mathias
 */
public class StrategyPair {
    private Strategy firstStrategy;
    private Strategy secondStrategy;
    
    private StrategyPair(Strategy firstStrategy, Strategy secondStrategy) {
        this.firstStrategy = firstStrategy;
        this.secondStrategy = secondStrategy;
    }
    
    public static StrategyPair fromTwoStrategies(Strategy firStrategy, Strategy seconStrategy) {
        return new StrategyPair(firStrategy, seconStrategy);
    }
    
    public boolean contains(Strategy strategy) {
        return getFirstStrategy().equals(strategy) || getSecondStrategy().equals(strategy);
    }
    
    @Override
    public String toString() {
        return firstStrategy + " + " + secondStrategy;
    }

    @Override
    public int hashCode() {
        int hash = 7;
        hash = 89 * hash + Objects.hashCode(this.firstStrategy.getName());
        hash += 89 * hash + Objects.hashCode(this.secondStrategy.getName());
        return hash;
    }
    
    @Override
    public boolean equals(Object obj) {
        if (obj == null) {
            return false;
        }
        if (getClass() != obj.getClass()) {
            return false;
        }
        final StrategyPair other = (StrategyPair) obj;
        if (!Objects.equals(this.firstStrategy.getName(), other.firstStrategy.getName())) {
            return false;
        }
        if (!Objects.equals(this.secondStrategy.getName(), other.secondStrategy.getName())) {
            return false;
        }
        return true;
    }
    

    public Strategy getFirstStrategy() {
        return firstStrategy;
    }

    public void setFirstStrategy(Strategy firstStrategy) {
        this.firstStrategy = firstStrategy;
    }

    public Strategy getSecondStrategy() {
        return secondStrategy;
    }

    public void setSecondStrategy(Strategy secondStrategy) {
        this.secondStrategy = secondStrategy;
    }
    
}
