package nash;

import helpers.ListHelper;
import java.util.ArrayList;
import java.util.Collections;
import java.util.HashMap;
import java.util.List;

/**
 *
 * @author Mathias
 */
public class NormalGame implements Game {

    private Matrix matrix;
    private final Player player1;
    private final Player player2;
    private HashMap<Strategy, Float> finalProbabilities;

    public NormalGame(Player player1, Player player2) {
        this.player1 = player1;
        this.player2 = player2;
        matrix = new Matrix(player1.getStrategies(), player2.getStrategies());
    }
    
    public NormalGame() {
        player1 = new Player("oben", "unten");
        player2 = new Player("links", "rechts");
        matrix = new Matrix(player1.getStrategies(), player2.getStrategies());
        initDefaultVectors();
    }

    private void initDefaultVectors() {
        for (int i = 0; i < 4; i++) {
            matrix.addField(new Vector(0, 0));
        }
    }

    public void setStrategies(List<Strategy> firstStrategies, List<Strategy> secondStrategies) {
        matrix = new Matrix(firstStrategies, secondStrategies);
    }

    public void addField(Vector payoff) {
        matrix.addField(payoff);
    }

    public void clear() {
        matrix.clear();
        finalProbabilities = null;
    }

    public Vector getVector(Strategy firstStrategy, Strategy secondStrategy) {
        return matrix.getVector(firstStrategy, secondStrategy);
    }

    public Strategy getStrategy(String name, Player player) {
        return matrix.getStrategy(name, player);
    }

    public void removeStrategy(String name, Player player) {
        if (player.getStrategies().size() == 1) {
            throw new IllegalStateException("Trying to remove last strategy from a player");
        }
        matrix.removeStrategy(name, player);
    }

    private List<Strategy> getOpponentStrategies(Player player) {
        if (player.equals(player1)) {
            return player2.getStrategies();
        } else {
            return player1.getStrategies();
        }
    }

    public List<Strategy> findDominatedStrategies(Player player) {  //Dann alle Matrix-Funktionen (Mehrspieler?).  Dann Flussdiagramm. Dann Javafx. Dann anfangen zu schreiben (?).
        List<Strategy> ownStrategies = player.getStrategies();
        List<Strategy> opponentStrategies = getOpponentStrategies(player);
        List<Strategy> dominatedStrategies = new ArrayList<>();
        for (Strategy ownStrategy : ownStrategies) {
            for (Strategy ownStrategy2 : ownStrategies) {
                if (ownStrategy.equals(ownStrategy2)) {
                    continue;
                }
                boolean weaklyDominated = false;
                List<Strategy> possiblyDominated = new ArrayList<>();
                for (Strategy opponentStrategy : opponentStrategies) {
                    if (payoffIsLess(player, ownStrategy, ownStrategy2, opponentStrategy)) {
                        possiblyDominated.add(ownStrategy);
                    } else if (payoffIsLess(player, ownStrategy2, ownStrategy, opponentStrategy)) {
                        possiblyDominated.add(ownStrategy2);
                    } else {
                        weaklyDominated = true;
                    }
                }
                if (ListHelper.isListOfSameElements(possiblyDominated)) {
                    Strategy dominatedStrat = possiblyDominated.get(0);
                    if (!dominatedStrategies.contains(dominatedStrat)) {
                        StrategyType type = weaklyDominated ? StrategyType.WeaklyDominated : StrategyType.Dominated;
                        dominatedStrat.setType(type);
                        dominatedStrategies.add(dominatedStrat);
                    }
                }
            }
        }
        return dominatedStrategies;
    }

    public boolean isZeroSumGame() {
        for (Vector vector : matrix.getVectors().values()) {
            if (vector.getFirst() + vector.getSecond() != 0) {
                return false;
            }
        }
        return true;
    }

    public ArrayList<NashEquilibrium> calculateAllNashEquilibria() {
        ArrayList<NashEquilibrium> equilibria = new ArrayList<>();
        equilibria.addAll(calculateNash());
        equilibria.add(calculateNashWithMixedStrategies());
        return equilibria;
    }

    @Override
    public ArrayList<NashEquilibrium> calculateNash() {
        ArrayList<StrategyPair> firstPairs = getPlayedStrategyPairs(player1);
        ArrayList<StrategyPair> secondPairs = getPlayedStrategyPairs(player2);
        firstPairs.retainAll(secondPairs);
        ArrayList<NashEquilibrium> nes = new ArrayList<>();
        for (StrategyPair pair : firstPairs) {
            nes.add(createEquilibrium(pair));
        }
        return nes;
    }

    private ArrayList<StrategyPair> getPlayedStrategyPairs(Player player) {
        List<Strategy> ownStrategies = player.getStrategies();
        List<Strategy> opponentStrategies = getOpponentStrategies(player);

        ArrayList<StrategyPair> playedStrategyPairs = new ArrayList<>();
        for (Strategy opponentStrategy : opponentStrategies) {
            float maximum = getMaximumPayoff(player, opponentStrategy);
            for (Strategy ownStrategy : ownStrategies) {
                float payoff = getPayoff(player, ownStrategy, opponentStrategy);
                if (payoff == maximum) {
                    if (player.equals(player1)) {
                        playedStrategyPairs.add(StrategyPair.fromTwoStrategies(ownStrategy, opponentStrategy));
                    } else {
                        playedStrategyPairs.add(StrategyPair.fromTwoStrategies(opponentStrategy, ownStrategy));
                    }
                }
            }
        }
        return playedStrategyPairs;
    }

    private float getPayoff(Player player, Strategy strategy1, Strategy strategy2) {
        Vector payoff = matrix.getVector(strategy1, strategy2);
        if (player.equals(player1)) {
            return payoff.getFirst();
        } else {
            return payoff.getSecond();
        }
    }

    @Override
    public NashEquilibrium calculateNashWithMixedStrategies() {
        if (finalProbabilities != null) {
            finalProbabilities.clear();
        }
        calculateProbabilities(player1);
        calculateProbabilities(player2);
        Strategy optimum1 = new Strategy("Optimal gemischt");
        Strategy optimum2 = new Strategy("Optimal gemischt");
        player1.getStrategies().add(optimum1);
        player2.getStrategies().add(optimum2);
        calculateNewMixedPayoffs(player1);
        calculateNewMixedPayoffs(player2);
        float aPayoff = getOptimalMixedPayoff(player1);
        float bPayoff = getOptimalMixedPayoff(player2);
        StrategyPair optimalPair = StrategyPair.fromTwoStrategies(optimum1, optimum2);
        matrix.getVectors().put(optimalPair, new Vector(aPayoff, bPayoff));
        return createEquilibrium(optimalPair);
    }

    private void calculateProbabilities(Player player) {
        HashMap<Strategy, Float> localProbabilities = calculateEvenProbabilities(player);
        if (finalProbabilities == null) {
            finalProbabilities = new HashMap<>();
        }
        for (int i = 0; i < 1000000; i++) {
            for (Strategy strat : localProbabilities.keySet()) {
                ArrayList<Float> payoffs = getMixedPayoffsOfOtherPlayer(localProbabilities, player);
                float difference = getMaxDifference(payoffs);
                if (difference == 0.0f || difference > -.01f && difference < .01f) {
                    for (Strategy probStrat : localProbabilities.keySet()) {
                        finalProbabilities.put(probStrat, round(localProbabilities.get(probStrat), 1));
                    }
                    return;
                } else {
                    localProbabilities = addProbability(localProbabilities, strat, .001f);
                    payoffs = getMixedPayoffsOfOtherPlayer(localProbabilities, player);
                    if (getMaxDifference(payoffs) > difference) {
                        localProbabilities = addProbability(localProbabilities, strat, -.002f);
                    }
                }
            }
        }
    }

    private void calculateNewMixedPayoffs(Player player) {
        if (finalProbabilities == null) {
            return;
        }
        List<Strategy> ownStrats = player.getStrategies();
        List<Strategy> otherStrats = getOpponentStrategies(player);
        for (Strategy otherStrat : otherStrats) {
            if (otherStrat.getName().equals("Optimal gemischt")) {
                continue;
            }
            float aPayoff = 0;
            float bPayoff = 0;
            for (Strategy ownStrat : ownStrats) {
                if (ownStrat.getName().equals("Optimal gemischt")) {
                    continue;
                }
                aPayoff += getPayoff(player1, ownStrat, otherStrat) * finalProbabilities.get(ownStrat) / 100;
                bPayoff += getPayoff(player2, ownStrat, otherStrat) * finalProbabilities.get(ownStrat) / 100;
            }
            aPayoff = round(aPayoff, 2);
            bPayoff = round(bPayoff, 2);
            if (player.equals(player1)) {
                matrix.getVectors().put(StrategyPair.fromTwoStrategies(ownStrats.get(ownStrats.size() - 1), otherStrat), new Vector(aPayoff, bPayoff));
            } else {
                matrix.getVectors().put(StrategyPair.fromTwoStrategies(otherStrat, ownStrats.get(ownStrats.size() - 1)), new Vector(aPayoff, bPayoff));
            }
        }
    }

    private float round(float value, int numberOfDecimals) {
        int decimalValue = (int) Math.pow(10, numberOfDecimals);
        return (float) Math.round(value * decimalValue) / decimalValue;
    }

    private float getOptimalMixedPayoff(Player player) {
        if (finalProbabilities == null) {
            return 0;
        }
        List<Strategy> ownStrats = player.getStrategies();
        List<Strategy> otherStrats = getOpponentStrategies(player);
        float payoff = 0;
        for (Strategy otherStrat : otherStrats) {
            if (otherStrat.getName().equals("Optimal gemischt")) {
                continue;
            }
            for (Strategy ownStrat : ownStrats) {
                if (ownStrat.getName().equals("Optimal gemischt")) {
                    continue;
                }
                float otherProbability = (finalProbabilities.get(otherStrat) / 100);
                float ownProbability = (finalProbabilities.get(ownStrat) / 100);
                payoff += otherProbability * ownProbability * getPayoff(player, ownStrat, otherStrat);
            }
        }
        return round(payoff, 1);
    }

    private ArrayList<Float> getMixedPayoffsOfOtherPlayer(HashMap<Strategy, Float> localProbabilities, Player player) {
        List<Strategy> ownStrategies = player.getStrategies();
        List<Strategy> otherStrategies = getOpponentStrategies(player);
        ArrayList<Float> payoffs = new ArrayList<>();
        for (Strategy otherStrategy : otherStrategies) {
            float payoff = 0;
            for (Strategy ownStrategy : ownStrategies) {
                if (player.equals(player1)) {
                    payoff += (matrix.getVector(ownStrategy, otherStrategy).getSecond()) * localProbabilities.get(ownStrategy);
                } else {
                    payoff += (matrix.getVector(ownStrategy, otherStrategy).getFirst()) * localProbabilities.get(ownStrategy);
                }
            }
            payoffs.add(payoff);
        }
        return payoffs;
    }
    
    public void print() {
        matrix.printVectors();
    }

    private float getMaximumPayoff(Player player, Strategy opponentStrategy) {
        List<Strategy> ownStrategies = player.getStrategies();
        ArrayList<Float> payoffs = new ArrayList<>();
        for (Strategy ownStrategy : ownStrategies) {
            payoffs.add(getPayoff(player, ownStrategy, opponentStrategy));
        }
        return Collections.max(payoffs);
    }

    private float getMaxDifference(ArrayList<Float> floats) {
        ArrayList<Float> differences = new ArrayList<>();
        for (int i = 0; i < floats.size(); i++) {
            for (int j = 0; j < floats.size(); j++) {
                differences.add(Math.abs(floats.get(i) - floats.get(j)));
            }
        }
        return Collections.max(differences);
    }

    private HashMap<Strategy, Float> calculateEvenProbabilities(Player player) {
        List<Strategy> ownStrats = player.getStrategies();
        HashMap<Strategy, Float> localProbabilities = new HashMap<>();
        for (Strategy ownStrat : ownStrats) {
            localProbabilities.put(ownStrat, (float) (1 / (float) ownStrats.size()) * 100);
        };
        return localProbabilities;
    }

    private HashMap<Strategy, Float> addProbability(HashMap<Strategy, Float> localProbabilities, Strategy strat, float percentage) {
        localProbabilities.put(strat, localProbabilities.get(strat) + percentage);
        for (Strategy s : localProbabilities.keySet()) {
            if (strat.equals(s)) {
                continue;
            }
            localProbabilities.put(s, localProbabilities.get(s) - percentage / (float) (localProbabilities.keySet().size() - 1));
        }
        return localProbabilities;
    }

    private NashEquilibrium createEquilibrium(StrategyPair strategyPair) {
        NashEquilibrium equilibrium = new NashEquilibrium(strategyPair.getFirstStrategy(), strategyPair.getSecondStrategy());
        NashEquilibriumType type = isStrict(equilibrium) ? NashEquilibriumType.Strict : NashEquilibriumType.NonStrict;
        equilibrium.setType(type);
        return equilibrium;
    }

    private boolean isStrict(NashEquilibrium equilibrium) {
        ArrayList<Float> firstPayoffs = getPayoffsForOpponentStrategy(player2, equilibrium.getFirstStrat());
        float payoff = Collections.max(firstPayoffs);
        if (Collections.frequency(firstPayoffs, payoff) > 1) {
            return false;
        }
        ArrayList<Float> secondPayoffs = getPayoffsForOpponentStrategy(player1, equilibrium.getSecondStrat());
        payoff = Collections.max(secondPayoffs);
        if (Collections.frequency(secondPayoffs, payoff) > 1) {
            return false;
        }
        return true;
    }

    private ArrayList<Float> getPayoffsForOpponentStrategy(Player player, Strategy opponentStrategy) {
        List<Strategy> ownStrats = player.getStrategies();
        ArrayList<Float> values = new ArrayList<>();
        for (Strategy ownStrat : ownStrats) {
            values.add(getPayoff(player, ownStrat, opponentStrategy));
        }
        return values;
    }

    private boolean payoffIsLess(Player player, Strategy firstStrategy, Strategy firstStrategy2, Strategy secondStrategy) {
        return getPayoff(player, firstStrategy, secondStrategy) < getPayoff(player, firstStrategy2, secondStrategy);
    }

    public Float getProbability(String strategyName, Player player) {
        if (finalProbabilities == null) {
            return null;
        }
        List<Strategy> ownStrategies = player.getStrategies();
        for (Strategy strat : ownStrategies) {
            if (strat.getName().equals(strategyName)) {
                return finalProbabilities.get(strat);
            }
        }
        return null;
    }
