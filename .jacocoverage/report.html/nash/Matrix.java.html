package nash;

import java.util.ArrayList;
import java.util.Collections;
import java.util.HashMap;
import java.util.List;

/**
 *
 * @author Mathias
 */
public class Matrix {

    private List<Strategy> firstPlayerStrats = new ArrayList<>();
    private List<Strategy> secondPlayerStrats = new ArrayList<>();
    private final HashMap<StrategyPair, Vector> vectors = new HashMap<>();
    private int verticalPosition = -1;
    private int horizontalPosition = 0;

    public Matrix(List<Strategy> firstPlayerStrats, List<Strategy> secondPlayerStats) {
        testStrategyNames(firstPlayerStrats);
        testStrategyNames(secondPlayerStats);
        this.firstPlayerStrats = firstPlayerStrats;
        this.secondPlayerStrats = secondPlayerStats;
    }

    public void printVectors() {
        for (Strategy fStrat : firstPlayerStrats) {
            for (Strategy sStrat : secondPlayerStrats) {
                System.out.print("Spieler 1: " + fStrat + "; Spieler 2: " + sStrat + " -> ");
                System.out.println(getVector(fStrat, sStrat));
            }
        }
    }

    public void addField(Vector payoff) {
        incrementPositions();
        vectors.put(StrategyPair.fromTwoStrategies(firstPlayerStrats.get(horizontalPosition), secondPlayerStrats.get(verticalPosition)), payoff);
    }

    public void removeStrategy(String name, Player player) {
        Strategy strategy = getStrategy(name, player);
        List<Strategy> strategies = player.getStrategies();
        strategies.remove(strategy);
        ArrayList<StrategyPair> toRemove = new ArrayList<>();
        for (StrategyPair pair : vectors.keySet()) {
            if (pair.contains(strategy)) {
                toRemove.add(pair);
            }
        }
        for (StrategyPair pair : toRemove) {
            vectors.remove(pair);
        }
    }

    private void incrementPositions() {
        int firstSize = firstPlayerStrats.size();
        int secondSize = secondPlayerStrats.size();
        if (verticalPosition + 1 < secondSize) {
            verticalPosition += 1;
        } else if (horizontalPosition + 1 < firstSize) {
            verticalPosition = 0;
            horizontalPosition += 1;
        }
    }
    
    private void testStrategyNames(List<Strategy> strategies) {
        ArrayList<String> names = new ArrayList<>();
        strategies.forEach(strat -> {
            names.add(strat.getName());
        });
        names.forEach(stratName -> {
            if (Collections.frequency(names, stratName) > 1) {
                throw new IllegalArgumentException("Same strategy twice!");
            }
        });
    }

    public void clear() {
        resetPositions();
        vectors.clear();
    }

    private void resetPositions() {
        verticalPosition = -1;
        horizontalPosition = 0;
    }

    public Vector getVector(Strategy firstStrat, Strategy secondStrat) {
        if (firstPlayerStrats.contains(firstStrat)) {
            return vectors.get(StrategyPair.fromTwoStrategies(firstStrat, secondStrat));
        } else {
            return vectors.get(StrategyPair.fromTwoStrategies(secondStrat, firstStrat));
        }
    }

    public Strategy getStrategy(String name, Player player) {
        List<Strategy> strategies = player.getStrategies();
        for (Strategy strategy : strategies) {
            if (strategy.getName().equals(name)) {
                return strategy;
            }
        }
        return null;
    }

    public HashMap<StrategyPair, Vector> getVectors() {
        return vectors;
    }
}
